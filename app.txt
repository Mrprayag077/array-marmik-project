const express = require('express');
const bodyParser = require('body-parser');
const { spawn } = require('child_process');

const app = express();
const port = 3000;

app.use(bodyParser.urlencoded({ extended: true }));
app.use(bodyParser.json());

// Route to handle image processing
app.post('/processImage', (req, res) => {
    // Get the image data from the request
    const imageData = req.body.imageData;

    // Spawn a Python process to run the script
    const pythonProcess = spawn('python', ['main.py', imageData]);

    // Listen for data from the Python script
    pythonProcess.stdout.on('data', (data) => {
        // Process the data received from the script
        const processedData = data.toString();

        // Send the processed data back to the frontend
        res.send({ processedData });
    });

    // Handle errors from the Python script
    pythonProcess.stderr.on('data', (data) => {
        console.error(`Error: ${data}`);
        res.status(500).send('Internal Server Error');
    });
});

app.listen(port, () => {
    console.log(`Server is running on port ${port}`);
});
